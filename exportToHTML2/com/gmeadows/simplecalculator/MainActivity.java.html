package com.gmeadows.simplecalculator; 
 
import android.support.v4.widget.TextViewCompat; 
import android.support.v7.app.AppCompatActivity; 
import android.os.Bundle; 
import android.view.View; 
 
public class MainActivity extends AppCompatActivity { 
    public String memory; 
    @Override 
    protected void onCreate(Bundle savedInstanceState) { 
        super.onCreate(savedInstanceState); 
        setContentView(R.layout.activity_main); 
        memory = ""; 
    } 
    public void pressedSeven(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "7"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedEight(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "8"; 
        tmp.setText(memory); 
    } 
 
    public void pressedAC(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory = "0"; 
        tmp.setText(memory); 
    } 
 
    public void pressedOne(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "1"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedTwo(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "2"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedThree(View view) {

        memory += "3"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedFour(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "4"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedFive(View view) { 
        TextView tmp = findViewById(R.id.result_text);

        tmp.setText(memory); 
 
    } 
 
    public void pressedSix(View view) { 
        TextView tmp = findViewById(R.id.result_text);

    } 
 
    public void pressedNine(View view) { 
        TextView tmp = findViewById(R.id.result_text);

    } 
 
    ////// 
    public void pressedPlus(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += ""; 
        tmp.setText(memory); 
 
    } 
 
    /////// 
    public void pressedMinus(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "7"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedNegative(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "7"; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedPercent(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory = result_text / 100; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedDivide(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory = result_text / nextDouble(); 
        tmp.setText(memory); 
 
    } 
 
    public void pressedMultiply(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory = result_text * nextDouble(); 
        tmp.setText(memory); 
 
    } 
 
    public void pressedDecimal(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        memory += "."; 
        tmp.setText(memory); 
 
    } 
 
    public void pressedEqual(View view) { 
        TextView tmp = findViewById(R.id.result_text); 
        tmp.setText(memory); 
 
    } 
}
